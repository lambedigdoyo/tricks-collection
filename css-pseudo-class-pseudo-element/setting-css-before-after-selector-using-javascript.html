Setting CSS Before/After Selector with Javascript 

<style>
* {
  box-sizing: border-box;
}
.pseudoBox {
  position: relative;
  width: 100%;
  height: 100px;
  margin: 10px auto;
  background-color: #555;
  color: #eee;
  }
</style>>

<div class=pseudoBox></div>

<script>
var cssToInsert = `.pseudoBox:before{content: 'Create pseudo-element before and after Selector with Javascript'; background-color: brown; margin: 5px; position: absolute; text-align: center; width: calc(100% - 22px); height: calc(100% - 12px); border: 1px solid #aaa; box-shadow: 0 0 12px #fff; border-radius: 8px; display: flex; align-items: center; justify-content: center; white-space: pre-wrap; font: bold 16px Arial; color: gold; text-shadow: 0.5px 0.5px 1px #000; padding: 0 5px}`;
var insertCSS = document.createElement("style");
insertCSS.innerHTML = cssToInsert;
document.head.insertAdjacentElement('beforeend', insertCSS);
</script>
